dilution_solver

dilution_solver

Code which helps with the preparation of solution samples en masse.

  1. If you don't have any sample concentrations prepared, but you know the concentration range for each compound that you would like to explore, then the Design of Experiments in the doe module can help with the creation of a set of sample concentrations.

  2. If you have a collection of sample concentrations to prepare, and you need to decide on which stock solutions to prepare, and how much to use for each sample, use the code in the routines module.

 15class Design:
 16    """
 17    Model:
 18        - *roots* are samples provided by the user (stock solutions)
 19        - *sources* are samples which can be used to create other samples by
 20        transferring liquids from them. Roots are sources, but samples can also
 21        be derived from them which are sources themselves (e.g. in serial
 22        dilutions).
 23        - *targets* are samples which are created from sources.
 24    """

 42        # (n_root, 2)
 43        self.root_bounds = root_bounds

 62    def volume_by_regression(self, sources, targets_conc, targets_vol):
 63        """
 64        Solving how much volume of a source solution is required to create a
 65        concentration of a target solution with a defined volume.
 66
 67        Solution via linear regression allows for easier use of mixtures as
 68        sources for dilution.
 69
 70        This method could fail if physically impossible values are given
 71        (i.e. the problem has no solution). Always validate the output.
 72        """
 73        # (n_compounds x n_sources) (n_sources x n_samples) = (n_compounds x n_samples)
 74        # Calculate moles required
 75        target_moles = targets_conc * targets_vol[:, np.newaxis]
 76
 77        # Regression of concentration of sources onto moles gives volumes
 78        self.reg.fit(sources.T, target_moles.T)
 79        transfer_volumes = self.reg.coef_.copy()  # (n_sources x n_samples)
 80
 81        return transfer_volumes
 82
 83    def calculate_source_transfer_volumes(self):
 84        """
 85        Calculate volumes required to make target concentrations, given
 86        source concentrations, then calculate the difference between the
 87        source volumes required and the target volumes to give the solvent
 88        required to make the correct solution concentration.
 89        """
 90        transfer_volumes = self.volume_by_regression(
 91            self.source_concentrations, self.target_concentrations, self.target_volumes
 92        )
 93        solvent_volumes = self.target_volumes - np.sum(transfer_volumes, axis=1)
 94        return transfer_volumes, solvent_volumes
 95
 96    def calculate_intermediate_transfer_volumes(self):
 97        """
 98        Calculate the volumes to be transferred from roots to all other sources.
 99        """
100
101        # Regress roots onto sources to calculate volumes to transfer
102        transfer_volumes = self.volume_by_regression(
103            self.root_concentrations, self.source_concentrations, self.source_volumes
104        )
105        solvent_volumes = self.source_volumes - np.sum(transfer_volumes, axis=1)
106        transfer_volumes = transfer_volumes[self.root_concentrations.shape[0] :]
107
108        # Create output accounting roots being part of sources (they should be
109        # removed by removing rows). Also, more sources have been added, so
110        # new columns must be added to the output.
111        root_dim = self.root_concentrations.shape[0]
112        source_dim = self.source_concentrations.shape[0]
113        add_dim_1 = transfer_volumes.shape[0]
114        add_dim_2 = source_dim - root_dim
115
116        transfer_volumes = np.hstack(
117            (transfer_volumes, np.zeros((add_dim_1, add_dim_2)))
118        )
119        solvent_volumes = solvent_volumes[self.root_volumes.shape[0] :]
120
121        return transfer_volumes, solvent_volumes

123    def validate_or_optimize(self) -> None:
124        """
125        Test if an experiment design will be feasible. If not, attempt to optimise
126        it so that it is feasible.
127        """
128
129        transfer_volumes, solvent_volumes = self.calculate_source_transfer_volumes()
130
131        # Find if any of the solvent volumes are less than one - indicates
132        # source solution needs to be updated.
133        if np.any(solvent_volumes < 0):
134            self.optimize_source_concentrations()

136    def optimize_source_concentrations(self) -> None:
137        """
138        Choose optimal source concentrations based on target
139        concentrations and volumes within bounds.
140        """
141
142        # create a copy to iterate over
143        temp_design = deepcopy(self)
144
145        # Optimize
146        ## Initial guess for concentrations
147        w0 = np.ones(temp_design.source_concentrations.shape[0])
148
149        weight_bounds = []
150        root_concs = self.get_root_concentrations()
151        root_bounds = self.root_bounds
152        for x in range(0, root_concs.shape[0]):
153            lower = 0.0
154            # max_conc = conc * max_w, max_w = max_conc / conc
155            upper = root_bounds[x][1] / np.amax(root_concs[x])
156            weight_bounds.append((lower, upper))
157
158        result = minimize(
159            evaluate_source_concentrations,
160            w0,
161            args=(temp_design),
162            bounds=weight_bounds,
163            method="SLSQP",
164        )
165
166        # Set optimized concentrations
167        new_root_concs = weighting_update(self.get_root_concentrations(), result.x)
168        self.set_root_concentrations(new_root_concs)
169
170        transfer_volumes, solvent_volumes = self.calculate_source_transfer_volumes()
171
172        # Check if the optimiser found a feasible result
173        if np.any(solvent_volumes < 0.0) or np.any(transfer_volumes < 0.0):
174            transfer_output = str(transfer_volumes).replace("\n", "\n\t")
175            raise ValueError(
176                f"Optimisation failed: negative volumes detected.\n"
177                f"Solvent addition volumes:\n\t{solvent_volumes}\n"
178                f"Target transfer volumes:\n\t{transfer_output}\n"
179                f"Consider increasing the bounds for optimisation of root solutions."
180            )

182    def serial_dilution(
183        self, possible_transfer_volumes, intermediate_volume=1.0
184    ) -> None:
185        """
186        Finds which volumes in stock_volumes is lower than the minimum volume which
187        can be transferred (min_volume), and creates new stock solutions which
188        allow for sample transfer to go ahead.
189        """
190
191        # Now check for volumes which are too small to transfer
192        min_volume = np.amin(possible_transfer_volumes)
193
194        # TODO: Account for solvent transfer volumes which are too low
195        transfer_volumes, solvent_volumes = self.calculate_source_transfer_volumes()
196
197        idx_transfer = np.where(transfer_volumes < min_volume)
198
199        num_intermediates = len([x for x in self.source_labels if "intermediate" in x])
200        for sample_idx, stock_idx in zip(idx_transfer[0], idx_transfer[1]):
201            # Get the lowest unfeasible stock volumes per stock
202            min_val = transfer_volumes[sample_idx, stock_idx]
203
204            # Calculate factor by which the stock_conc must be multiplied by
205            # for an addition of min_volume to work.
206            factor = min_val / min_volume
207
208            # Calculate the updated stock concentrations and add to
209            # new stock concentrations
210            stock_concs = self.get_source_concentrations()
211            intermediate_stock_conc = stock_concs[stock_idx] * factor
212            self.add_source(
213                intermediate_stock_conc,
214                intermediate_volume,
215                f"intermediate_{num_intermediates}",
216            )
217            num_intermediates += 1
218
219        root_volumes, intermediate_solvent_volumes = (
220            self.calculate_intermediate_transfer_volumes()
221        )
222
223        check_volumes = intermediate_solvent_volumes < 0.0
224        if np.any(check_volumes):
225            print("error")
226            problem_idx = np.where(check_volumes)[0]
227            print(f"Sample(s) {problem_idx} too high in volume.")
228            print("Sample volume update failed.")
229            print("The negative volumes in this array are the reason:")
230            print(intermediate_solvent_volumes)
231            print(
232                f"Consider increasing the concentrations of stock solutions other than {stock_idx}."
233            )
234            print()
